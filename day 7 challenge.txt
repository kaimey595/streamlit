import streamlit as st
import string

def check_password_strength(password):
    min_length = 8
    strength_score = 0
    reasons = []

    # Check length
    if len(password) >= min_length:
        strength_score += 1
    else:
        reasons.append("Password must be at least 8 characters long.")

    # Check for lowercase
    if any(char.islower() for char in password):
        strength_score += 1
    else:
        reasons.append("Add lowercase letters.")

    # Check for uppercase
    if any(char.isupper() for char in password):
        strength_score += 1
    else:
        reasons.append("Add uppercase letters.")

    # Check for digits
    if any(char.isdigit() for char in password):
        strength_score += 1
    else:
        reasons.append("Add numbers.")

    # Check for special characters
    if any(char in string.punctuation for char in password):
        strength_score += 1
    else:
        reasons.append("Add special characters (e.g. @, #, !).")

    # Determine strength level
    if strength_score <= 2:
        strength = "âŒ Weak"
        color = "red"
    elif strength_score == 3 or strength_score == 4:
        strength = "âš ï¸ Medium"
        color = "orange"
    else:
        strength = "âœ… Strong"
        color = "green"

    return strength, color, reasons
