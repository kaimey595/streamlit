import streamlit as st

# Utility functions
def is_positive(n):
    return n > 0

def is_even(n):
    return n % 2 == 0

def is_prime(n):
    if n <= 1:
        return False
    if n == 2:
        return True
    if n % 2 == 0:
        return False
    for i in range(3, int(n**0.5) + 1, 2):
        if n % i == 0:
            return False
    return True

# Streamlit UI
st.set_page_config(page_title="Number Checker", page_icon="🔢")

st.title("🔢 Number Checker")

number = st.number_input("Enter any number", step=1)

if number:
    st.markdown(f"**Positive:** {'Yes ✅' if is_positive(number) else 'No ❌'}")
    st.markdown(f"**Even:** {'Yes ✅' if is_even(number) else 'No ❌'}")
    st.markdown(f"**Prime:** {'Yes ✅' if is_prime(int(number)) else 'No ❌'}")
